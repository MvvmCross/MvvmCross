The steps to get the presenters working:

1. Add following method override to your MvxSetup class (usually Setup.cs/.fs):

	C#
	protected override IMvxAndroidViewPresenter CreateViewPresenter()
	{
		var presenter = new MvxFormsDroidPagePresenter();
		Mvx.RegisterSingleton<IMvxPageNavigationHost>(presenter);
		return presenter;
	}

	F#
	override __.CreateViewPresenter() =
	    let presenter = new MvxFormsDroidPagePresenter()
	    Mvx.RegisterSingleton<IMvxPageNavigationHost>(presenter)
	    presenter
